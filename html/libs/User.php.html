 15 class User {
 16   private $username = null;
 17   private $password = null;
 18   private $user_id = null;
 19   private $avatar = null;
 20   private $admin = null;
 21 
 22   function __construct($user_id='') {
 23     if($this->check_user_session()) {
 24       $this->set_username($_SESSION['username']);
 25       $this->set_user_id($_SESSION['user_id']);
 26     }
 27   }
 28 
 29   function get_user_id() {
 30     return $this->user_id;
 31   }
 32 
 33   function set_user_id($user_id) {
 34     $this->user_id = $user_id;
 35   }
 36 
 37   function get_username() {
 38     return $this->username;
 39   }
 40 
 41   function set_username($username) {
 42     $this->username = $username;
 43   }
 44 
 45   function get_avatar() {
 46     return $this->avatar;
 47   }
 48 
 49   function set_avatar($avatar) {
 50     $this->avatar = $avatar;
 51   }
 52 
 53   function set_admin($adm) {
 54     $this->admin = $adm;
 55   }
 56 
 57   function is_admin() {
 58     return $this->admin;
 59   }
 60 
 61   /**
 62    * This function will check a user submitted username and password against 
 63    * the database. If it exists, it sets up a new session
 64    */
 65   function login($username, $password) {
 66     $db = DB::get_instance();
 67     $sql = "SELECT * FROM user WHERE username = '$username' AND password = '$password'";
 68     $result= $db->query($sql);
 69     if ($db->count_rows($result)) {
 70       $row = $db->fetch_assoc($result);
 71       $this->set_user_id($row['id']);
 72       $this->set_username($row['username']);
 73       $this->set_avatar($row['avatar']);
 74       $this->set_admin($row['admin']);
 75       $this->create_user_session();
 76     }
 77   }
 78 
 79   // This function is used to start a session and set initial variables
 80   private function create_user_session() {
 81     session_start();
 82     $_SESSION['user_id'] = $this->get_user_id();
 83     $_SESSION['username'] = $this->get_username();
 84     $_SESSION['avatar'] = $this->get_avatar();
 85     $_SESSION['admin'] = $this->is_admin();
 86   }
 87 
 88   // This function checks to see if a user is logged in.
 89   function check_user_session() {
 90     session_start();
 91     if(array_key_exists('user_id', $_SESSION)
 92       && array_key_exists('username', $_SESSION)) {
 93       return true;
 94     }
 95     return false;
 96   }
 97 
 98 }
