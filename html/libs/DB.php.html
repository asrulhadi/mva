17 class DB {
18   /**
19    * This class provides the barest wrapper for MySQL
20    */
21 
22   private $host;
23   private $username;
24   private $password;
25   private $db_name;
26   private static $instance;
27 
28   private function __construct() {
29     global $db;
30     $this->host = $db['host'];
31     $this->username = $db['username'];
32     $this->password = $db['password'];
33     $this->db_name = $db['name'];
34     $this->connect();
35   }
36 
37   function connect() {
38     $this->connection = mysqli_connect($this->host, $this->username, $this->password);
39     if(!$this->connection) {
40       echo(mysqli_error());
41     }
42     mysqli_select_db($this->connection, $this->db_name);
43   }
44 
45   function query($sql) {
46     $result = mysqli_query($this->connection, $sql);
47     if(!$result) {
48       echo("Function query: " . mysqli_error($this->connection));
49     }
50     return $result;
51   }
52 
53   function count_rows($result) {
54     return mysqli_num_rows($result);
55   }
56 
57   function fetch_assoc($result) {
58     return mysqli_fetch_assoc($result);
59   }
60 
61   static function get_instance() {
62     if(!self::$instance) {
63       self::$instance = new DB;
64     }
65     return self::$instance;
66   }
67 
68 }
